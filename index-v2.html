// server.js import express from 'express'; import fetch from 'node-fetch'; import dotenv from 'dotenv'; dotenv.config(); // loads AIRTABLE_TOKEN from .env const app = express(); app.use(express.json()); const AIRTABLE_BASE = "appHwT3O8fnX0rS8d"; // your base const AIRTABLE_TABLE = "Patients"; app.post('/submit-optin', async (req, res) => { const { consent, timestamp } = req.body; try { const response = await fetch(`https://api.airtable.com/v0/${AIRTABLE_BASE}/${AIRTABLE_TABLE}`, { method: 'POST', headers: { 'Authorization': `Bearer ${process.env.AIRTABLE_TOKEN}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ fields: { Consent: consent, Timestamp: timestamp } }) }); if (!response.ok) throw new Error('Failed to save to Airtable'); res.status(200).send("Success"); } catch (err) { console.error(err); res.status(500).send("Error"); } }); app.listen(3000, () => console.log("Server running on port 3000"));
